52 /// Class for objects than can be advanced in time by an Explicit
53 /// Timestepper.
54 /// WARNING: For explicit time stepping to work the object's residual
55 /// function (as used by get_inverse_mass_matrix_times_residuals(..)) MUST
56 /// be in the form r = f(t, u) - [timestepper approximation to dudt]!
57 /// Standard implicit time stepping will work with plenty of residuals that
58 /// don't fit into this form. Some examples where implicit time stepping
59 /// will work fine but explicit will fail:
60 /// 1) The negation of the above formula, this implementation will end up
61 /// using dudt = - f(u,t).
62 /// 2) A residual which is implicit or non-linear in dudt, such as r = dudt
63 /// - u x dudt.
64 //===============================================================
65 class ExplicitTimeSteppableObject
66 {
67  //Dummy double value for time
68  static double Dummy_time_value;
69 public:
70 
71  ///Empty constructor
72  ExplicitTimeSteppableObject() { }
73 
74  /// Broken copy constructor
75  ExplicitTimeSteppableObject(const ExplicitTimeSteppableObject&)
76  {
77  BrokenCopy::broken_copy("ExplicitTimeSteppableObject");
78  }
79 
80  /// Broken assignment operator
81  void operator=(const ExplicitTimeSteppableObject&)
82  {
83  BrokenCopy::broken_assign("ExplicitTimeSteppableObject");
84  }
85 
86  ///Empty destructor
87  virtual ~ExplicitTimeSteppableObject() { }
88 
89  ///\short A single virtual function that returns the residuals
90  ///vector multiplied by the inverse mass matrix
91  virtual void get_dvaluesdt(DoubleVector &minv_res);
92 
93  /// Function that gets the values of the dofs in the object
94  virtual void get_dofs(DoubleVector &dofs) const;
95

97  virtual void get_dofs(const unsigned &t, DoubleVector &dofs) const;

100  virtual void set_dofs(const DoubleVector &dofs);
101 
102  ///Function that adds the values to the dofs
103  virtual void add_to_dofs(const double &lambda,
104  const DoubleVector &increment_dofs);
105 
106  /// \short Empty virtual function to do anything needed before a stage of
107  /// an explicit time step (Runge-Kutta steps contain multiple stages per
108  /// time step, most others only contain one).
109  virtual void actions_before_explicit_stage() {}
110 
111  /// \short Empty virtual function that should be overloaded to update any
112  /// dependent data or boundary conditions that should be advanced after each
113  /// stage of an explicit time step (Runge-Kutta steps contain multiple
114  /// stages per time step, most others only contain one).
115  virtual void actions_after_explicit_stage() {}
116 
117  /// \short Empty virtual function that can be overloaded to do anything
118  /// needed before an explicit step.
119  virtual void actions_before_explicit_timestep() {}
120 
121  /// \short Empty virtual function that can be overloaded to do anything
122  /// needed after an explicit step.
123  virtual void actions_after_explicit_timestep() {}
124 
125  ///\short Broken virtual function that should be overloaded to
126  ///return access to the local time in the object
127  virtual double &time();
128 
129  /// \short Virtual function that should be overloaded to return a pointer to a
130  /// Time object.
131  virtual Time* time_pt() const;
132 };
