65  // Storage for weights needs to be 2x(2 + 0/2) (the +2 is extra history

151  /// The "real" implementation of the implicit midpoint rule. Implemented
152  /// by calculation of residuals etc. at half step. This requires
153  /// non-trivial modifications to the element's residual and Jacobian
154  /// calculation functions to interpolate values to the midpoint. As such
155  /// IMRByBDF should be preferred.
156  ///
157  /// However this class must be used when multiple different time steppers
158  /// are being used simultaneously for different parts of the problem.
159 class IMR : public IMRBase
160 {
161 public:
162  /// Common mistakes when using this implementation of midpoint:
163  /// * Didn't include the d_u_evaltime_by_du_np1 term in the Jacobian.
164  /// * Didn't properly interpolate time/values/x/derivatives in
165  /// jacobian/residual (all of these MUST be evaluated at the midpoint).
166 
167 
168  /// Constructor with initialisation
169  IMR(const bool& adaptive=false) : IMRBase(adaptive) {}
170 
171  /// Destructor, predictor_pt handled by base
172  virtual ~IMR() {}
173 
174  /// Setup weights for time derivative calculations.
175  void set_weights()
176  {
177  // Set the weights for zero-th derivative (i.e. the value to use in
178  // newton solver calculations, implicit midpoint method uses the
179  // average of previous and current values).
180  Weight(0, 0) = 0.5;
181  Weight(0, 1) = 0.5;
182 
183  // Set the weights for 1st time derivative
184  double dt=Time_pt->dt(0);
185  Weight(1,0) = 1.0/dt;
186  Weight(1,1) = -1.0/dt;
187  }
188 
189  /// \short Number of history values to interpolate over to get the
190  /// "current" value.
191  unsigned nprev_values_for_value_at_evaluation_time() const {return 2;}
192 
193 private:
194 
195  /// Inaccessible copy constructor.
196  IMR(const IMR &dummy) {}
197 
198  /// Inaccessible assignment operator.
199  void operator=(const IMR &dummy) {}
200 };

203 /// Implementation of implicit midpoint rule by taking half a step of bdf1
204 /// then applying an update to all dofs. This implementation *should* work
205 /// with any existing problem for which the BDF methods work.
206 ///
207 /// The exception is when multiple different time steppers are being used
208 /// simultaneously for different parts of the problem. In this case the
209 /// IMR class should be used.
210 class IMRByBDF : public IMRBase
211 {
212 public:
213  /// Constructor with initialisation
214  IMRByBDF(const bool& adaptive=false) : IMRBase(adaptive)
215  {
216  Update_pinned = true;
217  }
218 
219  /// Destructor
220  virtual ~IMRByBDF() {}
221 
222  /// Setup weights for time derivative calculations.
223  void set_weights()
224  {
225  // Use weights from bdf1
226  double dt=Time_pt->dt(0);
227  Weight(1,0) = 1.0/dt;
228  Weight(1,1) = -1.0/dt;
229  }
230 
231  /// \short Number of history values to interpolate over to get the
232  /// "current" value. Evaluation time is the end of the bdf1 "half-step",
233  /// so only need one value as normal.
234  unsigned nprev_values_for_value_at_evaluation_time() const {return 1;}
235 
236  /// Half the timestep before starting solve
237  void actions_before_timestep(Problem* problem_pt);
238 
239  /// Take problem from t={n+1/2} to t=n+1 by algebraic update and restore
240  /// time step.
241  void actions_after_timestep(Problem* problem_pt);
242 
243  /// Should we update pinned variables after the half-step?
244  bool Update_pinned;
245 
246 private:
247 
248  /// Inaccessible copy constructor.
249  IMRByBDF(const IMRByBDF &dummy) {}
250 
251  /// Inaccessible assignment operator.
252  void operator=(const IMRByBDF &dummy) {}
