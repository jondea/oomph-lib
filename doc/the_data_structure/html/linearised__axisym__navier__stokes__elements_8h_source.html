281  /// \short Return the i-th component of du/dt at local node n.
282  /// Uses suitably interpolated value for hanging nodes.
283  double du_dt_linearised_axi_nst(const unsigned &n, const unsigned &i) const
284  {
285  // Get the data's timestepper
286  TimeStepper* time_stepper_pt = this->node_pt(n)->time_stepper_pt();
287 
288  // Initialise dudt
289  double dudt = 0.0;
290 
291  // Loop over the timesteps, if there is a non-steady timestepper
292  if (!time_stepper_pt->is_steady())
293  {
294  // Get the index at which the velocity is stored
295  const unsigned u_nodal_index = u_index_linearised_axi_nst(i);
296 
297  // Determine number of timsteps (past & present)
298  const unsigned n_time = time_stepper_pt->ntstorage();
299 
300  // Add the contributions to the time derivative
301  for(unsigned t=0;t<n_time;t++)
302  {
303  dudt += time_stepper_pt->weight(1,t)*nodal_value(t,n,u_nodal_index);
304  }
305  }
306 
307  return dudt;
308  }
