167  /// Compute second invariant of tensor
168  double second_invariant(const DenseMatrix<double>& tensor)
169  {
170  // get size of tensor
171  unsigned n=tensor.nrow();
172  double trace_of_tensor=0.0;
173  double trace_of_tensor_squared=0.0;
174  for (unsigned i=0; i<n; i++)
175  {
176  // Calculate the trace of the tensor: T_ii
177  trace_of_tensor += tensor(i,i);
178  // Calculate the trace of the tensor squared: T_ij T_ji
179  for (unsigned j=0; j<n; j++)
180  {
181  trace_of_tensor_squared += tensor(i,j)*tensor(j,i);
182  }
183  }
184 
185  // return the second invariant: 1.0/2.0*[(T_ii)^2 - T_ij T_ji]
186  return 0.5*(trace_of_tensor*trace_of_tensor-trace_of_tensor_squared);
187 
188  }

313  /// Max. # of Newton iterations
314  unsigned Max_iter=20;

316  /// Number of Newton iterations taken in most recent invocation
317  unsigned N_iter_taken=0;

320  /// documented (defaults to false)
321  bool Doc_Progress=false;

324  double FD_step=1.0e-8;

327  double Tol=1.0e-8;

329  /// Use steplength control do make globally convergent (default false)

332  /// \short Black-box FD Newton solver:
333  /// Calling sequence for residual function is
334  /// \code residual_fct(parameters,unknowns,residuals) \endcode
335  /// where all arguments are double Vectors.
336  /// unknowns.size() = residuals.size()
337  void black_box_fd_newton_solve(ResidualFctPt residual_fct,
338  const Vector<double>& params,
339  Vector<double>& unknowns)
340  {
341  // Jacobian, current and advanced residual Vectors
342  unsigned ndof=unknowns.size();
343  DenseDoubleMatrix jacobian(ndof);
344  Vector<double> residuals(ndof);
345  Vector<double> residuals_pls(ndof);
346  Vector<double> dx(ndof);
347  Vector<double> gradient(ndof);
348  Vector<double> newton_direction(ndof);
349 
350  double half_residual_squared=0.0;
351  double max_step=0.0;
352 
353  /// Reset number of Newton iterations taken in most recent invocation
354  N_iter_taken=0;
355 
356  // Newton iterations
357  for (unsigned iloop=0; iloop<Max_iter; iloop++)
358  {
359  // Evaluate current residuals
360  residual_fct(params,unknowns,residuals);
361 
362  // Get half of squared residual and find maximum step length
363  // for step length control
364  if (Use_step_length_control)
365  {
366  half_residual_squared=0.0;
367  double sum=0.0;
368  for (unsigned i=0; i<ndof; i++)
369  {
370  sum += unknowns[i]*unknowns[i];
371  half_residual_squared+=residuals[i]*residuals[i];
372  }
373  half_residual_squared*=0.5;
374  max_step=100.0*std::max(sqrt(sum),double(ndof));
375  }
376 
377 
378  // Check max. residuals
379  double max_res = std::fabs(*std::max_element(residuals.begin(),
380  residuals.end(),
381  AbsCmp<double>()));
382 
383 
384  // Doc progress?
385  if (Doc_Progress)
386  {
387  oomph_info << "\nNewton iteration iter=" << iloop
388  << "\ni residual[i] unknown[i] " << std::endl;
389  for (unsigned i=0; i<ndof; i++)
390  {
391  oomph_info << i << " " << residuals[i]
392  << " " << unknowns[i] << std::endl;
393  }
394  }
395 
396 
397  // Converged?
398  if (max_res<Tol)
399  {
400  return;
401  }
402 
403  // Next iteration...
404  N_iter_taken++;
405 
406  // FD loop for Jacobian
407  for (unsigned i=0; i<ndof; i++)
408  {
409  double backup=unknowns[i];
410  unknowns[i]+=FD_step;
411 
412  // Evaluate advanced residuals
413  residual_fct(params,unknowns,residuals_pls);
414 
415  // Do FD
416  for (unsigned j=0; j<ndof; j++)
417  {
418  jacobian(j,i)=(residuals_pls[j]-residuals[j])/FD_step;
419  }
420 
421  // Reset fd step
422  unknowns[i]=backup;
423  }
424 
425  if (Doc_Progress)
426  {
427  oomph_info << "\n\nJacobian: "<< std::endl;
428  jacobian.sparse_indexed_output(*(oomph_info.stream_pt()));
429  oomph_info << std::endl;
430  }
431 
432  // Get gradient
433  if (Use_step_length_control)
434  {
435  for (unsigned i=0; i<ndof; i++)
436  {
437  double sum=0.0;
438  for (unsigned j=0; j<ndof; j++)
439  {
440  sum += jacobian(j,i)*residuals[j];
441  }
442  gradient[i]=sum;
443  }
444  }
445 
446  // Solve
447  jacobian.solve(residuals,newton_direction);
448 
449  // Update
450  if (Use_step_length_control)
451  {
452  for (unsigned i=0; i<ndof; i++)
453  {
454  newton_direction[i]*=-1.0;
455  }
456  // Update with steplength control
457  Vector<double> unknowns_old(unknowns);
458  double half_residual_squared_old=half_residual_squared;
459  line_search(unknowns_old,
460  half_residual_squared_old,
461  gradient,
462  residual_fct,
463  params,
464  newton_direction,
465  unknowns,
466  half_residual_squared,
467  max_step);
468  }
469  else
470  {
471  // Direct Newton update:
472  for (unsigned i=0; i<ndof; i++)
473  {
474  unknowns[i]-=newton_direction[i];
475  }
476 
477  }
478 
479 
480  }
481 
482 
483  // Failed to converge
484  std::ostringstream error_stream;
485  error_stream<< "Newton solver did not converge in "
486  << Max_iter << " steps " << std::endl;
487 
488  throw OomphLibError(error_stream.str(),
489  OOMPH_CURRENT_FUNCTION,
490  OOMPH_EXCEPTION_LOCATION);
491  }

502  /// Line search helper for globally convergent Newton method
503  //=======================================================================
504  void line_search(const Vector<double>& x_old,
505  const double half_residual_squared_old,
506  const Vector<double>& gradient,
507  ResidualFctPt residual_fct,
508  const Vector<double>& params,
509  Vector<double>& newton_dir,
510  Vector<double>& x,
511  double& half_residual_squared,
512  const double& stpmax)
513  {
514  const double min_fct_decrease=1.0e-4;
515  double convergence_tol_on_x=1.0e-16;
516  double f_aux=0.0;
517  double lambda_aux=0.0;
518  double proposed_lambda=0.0;
519  unsigned n=x_old.size();
520  double sum=0.0;
521  for (unsigned i=0; i<n; i++)
522  {
523  sum += newton_dir[i]*newton_dir[i];
524  }
525  sum=sqrt(sum);
526  if (sum > stpmax)
527  {
528  for (unsigned i=0; i<n; i++)
529  {
530  newton_dir[i] *= stpmax/sum;
531  }
532  }
533  double slope=0.0;
534  for (unsigned i=0; i<n; i++)
535  {
536  slope += gradient[i]*newton_dir[i];
537  }
538  if (slope >= 0.0)
539  {
540  std::ostringstream error_stream;
541  error_stream << "Roundoff problem in lnsrch: slope=" << slope << "\n";
542  throw OomphLibError(error_stream.str(),
543  OOMPH_CURRENT_FUNCTION,
544  OOMPH_EXCEPTION_LOCATION);
545  }
546  double test=0.0;
547  for (unsigned i=0; i<n; i++)
548  {
549  double temp=std::fabs(newton_dir[i])/std::max(std::fabs(x_old[i]),1.0);
550  if (temp > test) test=temp;
551  }
552  double lambda_min=convergence_tol_on_x/test;
553  double lambda=1.0;
554  while (true)
555  {
556  for (unsigned i=0; i<n; i++)
557  {
558  x[i]=x_old[i]+lambda*newton_dir[i];
559  }
560 
561  // Evaluate current residuals
562  Vector<double> residuals(n);
563  residual_fct(params,x,residuals);
564  half_residual_squared=0.0;
565  for (unsigned i=0; i<n; i++)
566  {
567  half_residual_squared+=residuals[i]*residuals[i];
568  }
569  half_residual_squared*=0.5;
570 
571  if (lambda < lambda_min)
572  {
573  for (unsigned i=0; i<n; i++) x[i]=x_old[i];
574 
575  //Create an Oomph Lib warning
576  OomphLibWarning("Warning: Line search converged on x only!",
577  "BlackBoxFDNewtonSolver::line_search()",
578  OOMPH_EXCEPTION_LOCATION);
579  return;
580  }
581  else if (half_residual_squared <= half_residual_squared_old+
582  min_fct_decrease*lambda*slope)
583  {
584  return;
585  }
586  else
587  {
588  if (lambda == 1.0)
589  {
590  proposed_lambda = -slope/(2.0*(half_residual_squared-
591  half_residual_squared_old-slope));
592  }
593  else
594  {
595  double r1=half_residual_squared-
596  half_residual_squared_old-lambda*slope;
597  double r2=f_aux-half_residual_squared_old-lambda_aux*slope;
598  double a_poly=(r1/(lambda*lambda)-r2/(lambda_aux*lambda_aux))/
599  (lambda-lambda_aux);
600  double b_poly=(-lambda_aux*r1/(lambda*lambda)+lambda*r2/
601  (lambda_aux*lambda_aux))/(lambda-lambda_aux);
602  if (a_poly == 0.0)
603  {
604  proposed_lambda = -slope/(2.0*b_poly);
605  }
606  else
607  {
608  double discriminant=b_poly*b_poly-3.0*a_poly*slope;
609  if (discriminant < 0.0)
610  {
611  proposed_lambda=0.5*lambda;
612  }
613  else if (b_poly <= 0.0)
614  {
615  proposed_lambda=(-b_poly+sqrt(discriminant))/(3.0*a_poly);
616  }
617  else
618  {
619  proposed_lambda=-slope/(b_poly+sqrt(discriminant));
620  }
621  }
622  if (proposed_lambda>0.5*lambda)
623  {
624  proposed_lambda=0.5*lambda;
625  }
626  }
627  }
628  lambda_aux=lambda;
629  f_aux = half_residual_squared;
630  lambda=std::max(proposed_lambda,0.1*lambda);
631  }
632  }
633  }

1123  }
1124 
1125

1128  for (std::map<std::string, ArgInfo<bool> >::iterator it=
1129  Specified_command_line_flag.begin();
1130  it!=Specified_command_line_flag.end(); it++)
1131  {

1135  found_match=true;

1160  }
