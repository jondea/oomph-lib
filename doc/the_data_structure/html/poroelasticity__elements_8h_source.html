330  /// Calculate the FE representation of u
331  void interpolated_u(const Vector<double> &s,
332  Vector<double>& disp) const
333  {
334  //Find number of nodes
335  unsigned n_node = nnode();
336 
337  //Local shape function
338  Shape psi(n_node);
339 
340  //Find values of shape function
341  shape(s,psi);
342 
343  for(unsigned i=0;i<DIM;i++)
344  {
345  //Index at which the nodal value is stored
346  unsigned u_nodal_index = u_index(i);
347 
348  //Initialise value of u
349  disp[i] = 0.0;
350 
351  //Loop over the local nodes and sum
352  for(unsigned l=0;l<n_node;l++)
353  {
354  disp[i] += nodal_value(l,u_nodal_index)*psi[l];
355  }
356  }
357  }

387  /// Calculate the FE representation of q
388  void interpolated_q(const Vector<double> &s,
389  Vector<double> &u) const
390  {
391  unsigned n_q_basis = nq_basis();
392  unsigned n_q_basis_edge = nq_basis_edge();
393 
394  Shape q_basis(n_q_basis,DIM);
395 
396  get_q_basis(s,q_basis);
397  for(unsigned i=0;i<DIM;i++)
398  {
399  u[i]=0.0;
400  for(unsigned l=0;l<n_q_basis_edge;l++)
401  {
402  u[i] += q_edge(l)*q_basis(l,i);
403  }
404  for(unsigned l=n_q_basis_edge;l<n_q_basis;l++)
405  {
406  u[i] += q_internal(l-n_q_basis_edge)*q_basis(l,i);
407  }
408  }
409  }

434  /// Calculate the FE representation of div u
435  void interpolated_div_q(const Vector<double> &s, double &div_q) const
436  {
437  // Zero the divergence
438  div_q=0;
439 
440  // Get the number of nodes, q basis function, and q edge basis functions
441  unsigned n_node=nnode();
442  const unsigned n_q_basis = nq_basis();
443  const unsigned n_q_basis_edge = nq_basis_edge();
444 
445  // Storage for the divergence basis
446  Shape div_q_basis_ds(n_q_basis);
447 
448  // Storage for the geometric basis and it's derivatives
449  Shape psi(n_node);
450  DShape dpsi(n_node,DIM);
451 
452  // Call the geometric shape functions and their derivatives
453  this->dshape_local(s,psi,dpsi);
454 
455  // Storage for the inverse of the geometric jacobian (just so we can call
456  // the local to eulerian mapping)
457  DenseMatrix<double> inverse_jacobian(DIM);
458 
459  // Get the determinant of the geometric mapping
460  double det = local_to_eulerian_mapping(dpsi,inverse_jacobian);
461 
462  // Get the divergence basis (wrt local coords) at local coords s
463  get_div_q_basis_local(s,div_q_basis_ds);
464 
465  // Add the contribution to the divergence from the edge basis functions
466  for(unsigned l=0;l<n_q_basis_edge;l++)
467  {
468  div_q+=1.0/det*div_q_basis_ds(l)*q_edge(l);
469  }
470 
471  // Add the contribution to the divergence from the internal basis functions
472  for(unsigned l=n_q_basis_edge;l<n_q_basis;l++)
473  {
474  div_q+=1.0/det*div_q_basis_ds(l)*q_internal(l-n_q_basis_edge);
475  }
476  }

527  /// du/dt at local node n
528  double du_dt(const unsigned &n,
529  const unsigned &i) const
530  {
531  // Get the timestepper
532  TimeStepper* time_stepper_pt=node_pt(n)->time_stepper_pt();
533 
534  // Storage for the derivative - initialise to 0
535  double du_dt=0.0;
536 
537  // If we are doing an unsteady solve then calculate the derivative
538  if(!time_stepper_pt->is_steady())
539  {
540  // Get the nodal index
541  const unsigned u_nodal_index=u_index(i);
542 
543  // Get the number of values required to represent history
544  const unsigned n_time=time_stepper_pt->ntstorage();
545 
546  // Loop over history values
547  for(unsigned t=0;t<n_time;t++)
548  {
549  //Add the contribution to the derivative
550  du_dt+=time_stepper_pt->weight(1,t)*nodal_value(t,n,u_nodal_index);
551  }
552  }
553
