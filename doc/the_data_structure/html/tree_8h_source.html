55 /// A generalised tree base class that abstracts the common functionality
56 /// between the quad- and octrees used in mesh adaptation in two and
57 /// three dimensions, respectively.
58 ///
59 /// The tree can also be part of a forest. If that is the case, the root
60 /// of the tree will have pointers to the roots of neighbouring trees.
61 ///
62 /// The objects contained in the tree must be RefineableElements.
63 ///
64 /// The tree can be traversed and actions performed
65 /// at all its "nodes" or only at the leaf "nodes" ("nodes" without sons).
66 ///
67 /// Finally, the leaf "nodes" can be split depending on
68 /// a criteria defined by the object.
69 ///
70 /// Note that Trees are only generated by splitting existing
71 /// Trees. Therefore, the constructors are protected. The
72 /// only Tree that "Joe User" can create is
73 /// the (derived) class TreeRoot.
74 //=================================================================

294 /// TreeRoot is a Tree that forms the root of a (recursive)
295 /// tree. The "root node" is special as it holds additional
296 /// information about its neighbours and their relative
297 /// rotation (inside a TreeForest).
298 //==================================================================
299 class TreeRoot : public virtual Tree
300 {
301  protected:
302 
303  /// \short Map of pointers to the neighbouring TreeRoots:
304  /// Neighbour_pt[direction] returns the pointer to the
305  /// TreeRoot's neighbour in the (enumerated) direction.
306  /// Returns NULL if there's no neighbour in this direction.
307  std::map<int,TreeRoot*> Neighbour_pt;
308 
309 
310  /// \short Map of booleans used for periodic boundaries:
311  /// Neighbour_periodic_direction[directon] returns true if the
312  /// neighbour in that direction is actually a periodic neighbour
313  /// --- shared data values, but independent position.
314  /// The default return of the map is false.
315  std::map<int,bool> Neighbour_periodic;
316 
317  public:
318 
319  ///Constructor for the (empty) root tree
320  TreeRoot(RefineableElement* const &object_pt) : Tree(object_pt)
321  {
322  //TreeRoot is the Root
323  Root_pt = this;
324  }

339  /// \short Return the pointer to the neighbouring TreeRoots in specified
340  /// direction. Returns NULL if there's no neighbour in this direction.
341  TreeRoot* &neighbour_pt(const int &direction)
342  {return Neighbour_pt[direction];}
343 
344  /// \short Return whether the neighbour in the particular direction
345  /// is periodic.
346  bool is_neighbour_periodic(const int &direction)
347  {return Neighbour_periodic[direction];}
348

350  void set_neighbour_periodic(const int &direction)
351  {Neighbour_periodic[direction] = true;}
352

354  void set_neighbour_nonperiodic(const int &direction)
355  {Neighbour_periodic[direction] = false;}
356 
357  ///Return the number of neighbours
358  unsigned nneighbour()
359  {
360  //Loop over the neighbours and test whether they are non-null
361  unsigned count=0;
362  for(std::map<int,TreeRoot*>::iterator it=Neighbour_pt.begin();
363  it!=Neighbour_pt.end();it++)
364  {
365  if(Neighbour_pt[it->first]!=0) {count++;}
366  }
367  //Return number counted
368  return count;
369  }
